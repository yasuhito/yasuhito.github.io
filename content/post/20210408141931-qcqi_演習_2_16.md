\begin{eqnarray}
  P^2 &=& \sum\_i |i \rangle\langle i| \sum\_j |j \rangle\langle j| \newline
    &=& \sum\_{i,j} |i \rangle\langle i|j \rangle\langle j| \newline
    &=& \sum\_{i,j} |i \rangle\delta\_{i,j}\langle j| \newline
    &=& \sum\_i |i \rangle\langle i| \newline
    &=& P
\end{eqnarray}
