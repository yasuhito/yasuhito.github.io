\begin{eqnarray\*}
[A, B]^{\dagger} &=& (AB - BA)^{\dagger} \\\\\\
\end{eqnarray\*}

[QCQI/演習 2.14]({{< relref "20210408102838-qcqi_演習_2_14" >}}) より

\begin{eqnarray\*}
\phantom{[A, B]^{\dagger}} &=& (AB)^{\dagger} - (BA)^{\dagger} \\\\\\
  &=& B^{\dagger}A^{\dagger} - A^{\dagger}B^{\dagger} \\\\\\
  &=& [B^{\dagger}, A^{\dagger}]
\end{eqnarray\*}
